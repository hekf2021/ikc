var gulp        = require('gulp');
var browserSync = require('browser-sync').create();

// 静态服务器 + 监听 scss/html 文件
gulp.task('serve', function() {

    browserSync.init({
        server: "./app"
    });
    gulp.watch("app/*.html").on('change', browserSync.reload);
});
gulp.task('default', ["serve"]);
